413double LosslessCodec::calculate_entropy(Mat mat) {
414
415 int bins[256] = {0};
416 int val;
417 for(int i = 0; i < mat.size().height; i++) {
418 for (int j = 0; j < mat.size().width; j++) {
419 val = (int) mat.at<uchar>(i,j);
420 bins[val]++;
421 }
422 }
423
424 double size = mat.size().width * mat.size().height;
425 double entropy = 0;
426 for(int i = 0; i < 256; i++) {
427 if (bins[i] > 0) {
428 entropy += (bins[i]/size) * (log(bins[i]/size));
429 }
430 }
431
432 return entropy*(-1);
